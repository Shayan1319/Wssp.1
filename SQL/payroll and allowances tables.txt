CREATE TABLE allowances (
    id INT AUTO_INCREMENT PRIMARY KEY,
    allowance VARCHAR(100) NOT NULL,
    fin_classification varchar(50) NOT NULL,
    rate_calc_mode varchar(50) NOT NULL,
    earning_deduction_fund varchar(50) NOT NULL,
    allowance_status tinyint NOT NULL default 1
);


create table earning_deduction_fund (
	`id` int not null AUTO_INCREMENT PRIMARY KEY,
	`employee_id` int not null,
	`fund` int not null,
	`gross_pay` int not null,
	`deduction` int not null,
	`net_pay` int not null,
	constraint `fk_earning_deduction_fund_employee_id` FOREIGN KEY (`employee_id`) REFERENCES employeedata(`id`)
)engine=INNODB;

CREATE TABLE rate (
  `id` int NOT NULL AUTO_INCREMENT PRIMARY KEY,
  `rate` varchar(50) not null,
  `employee_id` int NOT NULL,
  `allowances_id` int NOT NULL,
  CONSTRAINT `fk_rate_employee_id` FOREIGN KEY (`employee_id`) REFERENCES employeedata(`id`),
  CONSTRAINT `fk_rate_allowance_id` FOREIGN KEY (`allowances_id`) REFERENCES allowances(`id`)
) ENGINE=INNODB;
